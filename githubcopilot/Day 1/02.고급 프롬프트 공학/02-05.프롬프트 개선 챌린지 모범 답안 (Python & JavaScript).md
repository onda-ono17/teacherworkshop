## 1. 챌린지 1: 구체성(Specificity) 개선 결과

### [Python 모범 답안]
import re

class InvalidEmailError(Exception):
    """이메일 형식이 유효하지 않을 때 발생하는 커스텀 예외"""
    pass

def validate_user_email(email: str) -> bool:
    # 1. 정규표현식을 사용한 형식 검증
    email_regex = r'^[a-zA-Z0-9+-_.]+@[a-zA-Z0-9-]+\.[a-zA-Z0-9-.]+$'
    if not re.match(email_regex, email):
        raise InvalidEmailError("유효하지 않은 이메일 형식입니다.")
    
    # 2. 특정 도메인(example.com) 제한
    domain = email.split('@')[-1]
    if domain == 'example.com':
        raise InvalidEmailError("example.com 도메인은 사용할 수 없습니다.")
    
    return True

### [JavaScript 모범 답안]
class InvalidEmailError extends Error {
  constructor(message) {
    super(message);
    this.name = "InvalidEmailError";
  }
}

/**
 * 이메일 유효성을 검사하고 결과를 반환합니다.
 */
function validateUserEmail(email) {
  // 1. 정규표현식 검증
  const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  if (!emailRegex.test(email)) {
    throw new InvalidEmailError("유효하지 않은 이메일 형식입니다.");
  }

  // 2. 특정 도메인 제한
  const domain = email.split('@').pop();
  if (domain === 'example.com') {
    throw new InvalidEmailError("example.com 도메인은 사용할 수 없습니다.");
  }

  return true;
}

## 2. 챌린지 2: 맥락(Context) 활용 결과

### [Python (SQLAlchemy 기준)]
from sqlalchemy.orm import Session
from models import User  # 열려 있는 탭의 User 클래스 참조

def save_new_user(db: Session, email: str, username: str):
    # models.py의 구조를 참조하여 객체 생성
    new_user = User(
        email=email,
        username=username
    )
    
    db.add(new_user)
    db.commit()
    db.refresh(new_user)
    return new_user

## 3. 챌린지 3: 패러다임(Paradigm) 적용 결과

### [Python (FastAPI + Pydantic)]
from fastapi import FastAPI, Depends, HTTPException
from sqlalchemy.orm import Session
from typing import List
from database import get_db
from schemas import UserResponse
from models import User

app = FastAPI()

@app.get("/users", response_model=List[UserResponse])
async def read_users(db: Session = Depends(get_db)):
    # 1. async 비동기 방식 준수
    # 2. Depends를 통한 의존성 주입
    # 3. response_model을 통한 반환 타입 지정
    users = db.query(User).all()
    return users

### [JavaScript (Express + Async/Await)]
const express = require('express');
const router = express.Router();
const { User } = require('./models');

// 비동기 핸들러와 명확한 상태 코드 반환
router.get('/users', async (req, res, next) => {
  try {
    const users = await User.findAll();
    // HTTP 200 상태 코드와 데이터 반환
    res.status(200).json(users);
  } catch (error) {
    next(error);
  }
});

## 실습 요약 및 평가
개선된 프롬프트(After)를 사용하면 GitHub Copilot은 다음과 같은 특징을 보입니다:
1. 커스텀 예외 처리와 같은 방어적 코딩 패턴을 자동으로 포함합니다.
2. 프로젝트에 이미 존재하는 클래스(User)나 함수 정보를 정확히 인용합니다.
3. 특정 프레임워크(FastAPI 등)의 고유한 문법(Depends 등)을 올바르게 구사합니다.
